      setTimeout(() => {

                          </p>
                        </CardContent>
                      </Card>
                    ) : (
                      <Card className="border-white/10 bg-black/20">
                        <CardContent className="py-8 text-center">
                          <Shield className="mx-auto mb-4 h-12 w-12 text-white/40" />
                          <p className="text-white/70">
                            Select a support champion to see detailed matchup
                            tips and build recommendations.
                          </p>
                        </CardContent>
                      </Card>
                    )}
                  </TabsContent>

                  <TabsContent value="ally" className="space-y-6">
                    <div>
                      <h3 className="mb-4 text-lg font-semibold text-blue-300">
                        ADC Champions
                      </h3>
                      <div className="mb-6 grid grid-cols-6 gap-3 sm:grid-cols-8 md:grid-cols-10 lg:grid-cols-12">
                        {ADC_CHAMPIONS.map((champion) => (
                          <ChampionImage
                            key={champion}
                            championName={champion}
                            isSelected={selectedADC === champion}
                            onClick={() => {
                              setSelectedADC(
                                selectedADC === champion ? '' : champion
                              );
                              setSelectedSupport('');
                            }}
                          />
                        ))}
                      </div>
                    </div>

                    {selectedADC && adcMatchup ? (
                      <Card className="border-white/10 bg-black/20">
                        <CardHeader className="pb-3">
                          <CardTitle className="flex items-center justify-between text-lg text-white">
                            <div className="flex items-center gap-3">
                              <ChampionImage
                                championName={selectedADC}
                                isSelected={true}
                                size={40}
                                interactive={false}
                                showLabel={false}
                              />
                              <span>
                                Playing With {formatChampionName(selectedADC)}
                              </span>
                            </div>
                            <Badge
                              className={`border ${getSynergyColor(adcMatchup.synergy)}`}
                            >
                              {adcMatchup.synergy} Synergy
                            </Badge>
                          </CardTitle>
                        </CardHeader>
                        <CardContent className="space-y-4">
                          <div>
                            <h4 className="mb-2 font-semibold text-purple-300">
                              Synergy Tips:
                            </h4>
                            <ul className="space-y-1 text-sm text-white/80">
                              {adcMatchup.tips.map((tip, index) => (
                                <li
                                  key={index}
                                  className="flex items-start gap-2"
                                >
                                  <span className="mt-1 text-purple-400">
                                    •
                                  </span>
                                  <span>{tip}</span>
                                </li>
                              ))}
                            </ul>
                          </div>
                          <div>
                            <h4 className="mb-2 font-semibold text-blue-300">
                              Playstyle:
                            </h4>
                            <p className="text-sm text-white/80">
                              {adcMatchup.playstyle || 'General'}
                            </p>
                          </div>
                          {adcMatchup.optimalAttachTargets ? (
                            <div>
                              <h4 className="mb-2 font-semibold text-sky-300">
                                Optimal Attach Targets:
                              </h4>
                              <p className="text-sm text-white/80">
                                {adcMatchup.optimalAttachTargets}
                              </p>
                            </div>
                          ) : null}
                          {adcMatchup.buildAdjustments?.length ? (
                            <div>
                              <h4 className="mb-2 font-semibold text-emerald-300">
                                Build Considerations with this ADC:
                              </h4>
                              <ul className="space-y-1 text-sm text-white/80">
                                {adcMatchup.buildAdjustments.map(
                                  (adjustment) => (
                                    <li
                                      key={adjustment}
                                      className="flex items-start gap-2"
                                    >
                                      <span className="mt-1 text-emerald-300">
                                        •
                                      </span>
                                      <span>{adjustment}</span>
                                    </li>
                                  )
                                )}
                              </ul>
                              <p className="mt-1 text-xs text-white/60">
                                These reflect what your ADC likes. Always weigh
                                this against enemy threats (e.g., take
                                Mikael&apos;s vs heavy CC even if the synergy
                                suggests Mandate). No global first-item rule.
                              </p>
                            </div>
                          ) : null}
                        </CardContent>
                      </Card>
                    ) : selectedADC ? (
                      <Card className="border-white/10 bg-black/20">
                        <CardContent className="py-8 text-center">
                          <p className="text-white/70">
                            Detailed matchup data for{' '}
                            {formatChampionName(selectedADC)} is not available
                            yet. Use general Yuumi strategies with this ADC.
                          </p>
                        </CardContent>
                      </Card>
                    ) : (
                      <Card className="border-white/10 bg-black/20">
                        <CardContent className="py-8 text-center">
                          <Users className="mx-auto mb-4 h-12 w-12 text-white/40" />
                          <p className="text-white/70">
                            Select an ADC champion to see detailed synergy tips
                            and playstyle recommendations.
                          </p>
                        </CardContent>
                      </Card>
                    )}
                  </TabsContent>
                </Tabs>
              </CardContent>
            </Card>
          </TabsContent>

          {/* Skins Tab */}
          <TabsContent value="skins" className="space-y-6">
            <Card className="border-white/10 bg-black/30 backdrop-blur">
              <CardHeader>
                <CardTitle className="flex items-center gap-2 text-white">
                  <Palette className="h-5 w-5" />
                  Yuumi&apos;s Skins Collection
                </CardTitle>
              </CardHeader>
              <CardContent className="text-white/90">
                <div className="grid grid-cols-1 gap-6 md:grid-cols-2 lg:grid-cols-3">
                  {YUUMI_SKINS.map((skin) => (
                    <Card
                      key={skin.id}
                      className="overflow-hidden border-white/10 bg-black/20"
                    >
                      <div className="relative aspect-[4/3] bg-gradient-to-br from-purple-500/20 to-blue-500/20">
                        <SkinSplash src={skin.splashart} alt={skin.name} />
                        {skin.releaseDate === 'TBD' && (
                          <div className="absolute left-2 top-2">
                            <Badge
                              variant="outline"
                              className="border-2 border-yellow-400 bg-black/70 text-yellow-300 backdrop-blur-sm"
                            >
                              Unreleased
                            </Badge>
                          </div>
                        )}
                        <div className="absolute right-2 top-2">
                          <Badge
                            variant="outline"
                            className={`border-2 bg-black/70 backdrop-blur-sm ${
                              skin.availability === 'Permanent'
                                ? 'border-green-400 text-green-300'
                                : skin.availability === 'Prestige'
                                  ? 'border-yellow-400 text-yellow-300'
                                  : 'border-red-400 text-red-300'
                            }`}
                          >
                            {skin.availability}
                          </Badge>
                        </div>
                        {skin.previewUrl && (
                          <div className="absolute bottom-2 left-2">
                            <Button
                              size="sm"
                              onClick={() => setSelectedVideo(skin.previewUrl!)}
                              className="h-7 border-0 bg-black/60 px-2 text-xs text-white hover:bg-black/80"
                            >
                              <Play className="mr-1 h-3 w-3" />
                              Preview
                            </Button>
                          </div>
                        )}
                      </div>
                      <CardContent className="p-4">
                        <h3 className="mb-2 font-semibold text-white">
                          {skin.name}
                        </h3>
                        <div className="space-y-1 text-sm text-white/70">
                          <div className="flex justify-between">
                            <span>Release:</span>
                            <span>
                              {skin.releaseDate === 'TBD'
                                ? 'TBD (Unreleased)'
                                : skin.releaseDate}
                            </span>
                          </div>
                          <div className="flex justify-between">
                            <span>Price:</span>
                            <span>
                              {skin.rpPrice === 0
                                ? 'Included'
                                : `${skin.rpPrice} ${skin.currency === 'ME' ? 'ME' : 'RP'}`}
                            </span>
                          </div>
                          {skin.lastInShop && (
                            <div className="flex justify-between">
                              <span>Last in Shop:</span>
                              <span>{skin.lastInShop}</span>
                            </div>
                          )}
                          {skin.availability !== 'Permanent' &&
                            skin.expectedReturn && (
                              <div className="flex justify-between">
                                <span>Expected Return:</span>
                                <span>{skin.expectedReturn}</span>
                              </div>
                            )}
                          {/* Availability badge moved to splash overlay */}
                        </div>
                      </CardContent>
                    </Card>
                  ))}
                </div>

                <div className="mt-6 rounded-lg border border-blue-500/20 bg-blue-500/10 p-4">
                  <h3 className="mb-2 text-lg font-semibold text-blue-300">
                    Skin Information
                  </h3>
                  <div className="space-y-2 text-sm text-white/80">
                    <p>
                      <strong>Permanent skins:</strong> Always available in the
                      store with occasional discounts.
                    </p>
                    <p>
                      <strong>Legacy/Limited skins:</strong> Return during
                      specific events or limited-time reruns (e.g., Bewitching
                      during Halloween, Worlds event skins during Worlds).
                    </p>
                    <p>
                      <strong>Prestige skins:</strong> Available via Mythic
                      Essence and occasionally in Mythic Shop rotations.
                    </p>
                    <p>
                      <strong>Discount schedule:</strong> Most skins get
                      discounted around holidays and special events. Check the
                      League client for current sales.
                    </p>
                  </div>
                </div>
              </CardContent>
            </Card>

            {/* Video Modal */}
            {selectedVideoId && (
              <div
                className="fixed inset-0 z-50 flex items-center justify-center bg-black/80 backdrop-blur-sm"
                onClick={(e) => {
                  if (e.target === e.currentTarget) setSelectedVideo(null);
                }}
              >
                <div className="relative mx-4 w-full max-w-4xl">
                  <div className="overflow-hidden rounded-lg bg-black/90">
                    <div className="flex items-center justify-between border-b border-white/10 p-4">
                      <h3 className="text-lg font-semibold text-white">
                        Skin Preview
                      </h3>
                      <Button
                        variant="ghost"
                        size="sm"
                        onClick={() => setSelectedVideo(null)}
                        className="text-white hover:bg-white/10"
                      >
                        <X className="h-4 w-4" />
                      </Button>
                    </div>
                    <div className="aspect-video">
                      <iframe
                        src={`https://www.youtube.com/embed/${selectedVideoId}`}
                        title="Skin Preview"
                        className="h-full w-full"
                        allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture"
                        allowFullScreen
                      />
                    </div>
                  </div>
                </div>
              </div>
            )}
          </TabsContent>
        </Tabs>

        {/* Footer */}
        <div className="mt-12 flex items-center justify-between text-xs text-white/50">
          <span>
            Comprehensive Yuumi Guide • Updated for Patch {PATCH} • Data
            verified from OP.GG, U.GG, LoLalytics
          </span>
          <div className="flex items-center gap-4">
            <Link
              href="/gallery"
              className="underline/20 hover:text-white hover:underline"
            >
              Rule GIF Gallery
            </Link>
          </div>
        </div>
      </div>
    </div>
  );
}
